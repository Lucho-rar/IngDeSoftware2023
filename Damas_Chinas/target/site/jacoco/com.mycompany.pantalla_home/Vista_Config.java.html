    private void boton_mutear_desmutearActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_boton_mutear_desmutearActionPerformed
        // TODO add your handling code here:
        if (estaMute==false){
            label_enabled.setVisible(false);
            label_disabled.setVisible(true);
            estaMute=true;
            this.m.disperformMusic();
          
        }else{
            label_enabled.setVisible(true);
            label_disabled.setVisible(false);
            estaMute=false;
            this.m.performMusic();
        }
    }//GEN-LAST:event_boton_mutear_desmutearActionPerformed
